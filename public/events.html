This endpoint allows an authenticated user to create a new event. Images are uploaded using Multer middleware.


Request: 
{
  "title": "New Event",
  "description": "Description of the event.",
  "location": "Venue Address",
  "date": "2024-12-25",
  "start_time": "10:00 AM",
  "end_time": "02:00 PM",
  "is_paid": true,
  "ticket_price": 100,
  "max_attendees": 200,
  "category": "Conference",
  "images": [File uploads]
}

Response:
{
  "success": true,
  "data": {
    "id": "60a77a8c8f08b7fabc123456",
    "event_id": "12345",
    "eventAuthor": "John Doe",
    "title": "New Event",
    "description": "Description of the event.",
    "location": "Venue Address",
    "date": "2024-12-25",
    "startTime": "10:00 AM",
    "endTime": "02:00 PM",
    "isPaid": true,
    "ticketPrice": 100,
    "maxAttendees": 200,
    "category": "Conference",
    "images": ["img url", "img url"]
  }
}
